function trackShipment() { const input = document.querySelector('.tracking-input').value.trim(); const resultDiv = document.getElementById('tracking-result'); const validWaybill = ["GV-6460748-US", "XY646071448"]; if (input === validWaybill) { // 1. Check if we have seen this device before let firstTrackDate = localStorage.getItem('tracking_start_date'); if (!firstTrackDate) { // First time tracking! Save today's date firstTrackDate = new Date().getTime(); localStorage.setItem('tracking_start_date', firstTrackDate); } // 2. Calculate time difference const now = new Date().getTime(); const twoDaysInMilliseconds = 2 * 24 * 60 * 60 * 1000; const timeElapsed = now - firstTrackDate; // 3. Default (Under 2 days): Europe Hub let statusTitle = "IN TRANSIT"; let locationText = "Package is currently at the distribution center in Europe Hub."; let progressWidth = "65%"; let lastEvent = "Departed International Sort Facility"; let displayDate = "April 20, 2026"; // Initial Date // 4. Update (Over 2 days): JFK Airport if (timeElapsed >= twoDaysInMilliseconds) { statusTitle = "ARRIVED AT HUB"; locationText = "Package has arrived at JFK Airport, New York. Customs clearance in progress."; progressWidth = "90%"; lastEvent = "Arrived at Destination Sort Facility - New York (JFK)"; displayDate = "April 22, 2026"; // Updated Date } resultDiv.innerHTML = `
Status: ${statusTitle} ID: ${validWaybill}

${locationText}

Last Event: ${lastEvent}

${displayDate} - Updated Recently

`; } else { resultDiv.innerHTML = "

Waybill number not recognized.

"; } }
